use scripting additions

property indexFileName : "index.md"
property coverFileName : "cover.jpg"

on padded(n, digits)
	set s to n as string
	repeat while (length of s < digits)
		set s to "0" & s
	end repeat
	return s
end padded

on todayStamp()
	set d to current date
	set y to year of d as integer
	set m to month of d as integer
	set dd to day of d as integer
	return (y as string) & padded(m, 2) & padded(dd, 2)
end todayStamp

on todayYMD()
	set d to current date
	set y to year of d as integer
	set m to month of d as integer
	set dd to day of d as integer
	return (y as string) & "-" & padded(m, 2) & "-" & padded(dd, 2)
end todayYMD

on nextSequenceInParent(parentGroup, dateStr)
	tell application id "DNtp"
		set existing to children of parentGroup
		set maxSeq to 0
		repeat with r in existing
			try
				set nm to name of r as string
				if nm starts with (dateStr & "-") then
					set restName to text ((length of dateStr) + 2) thru -1 of nm
					set digitsOnly to ""
					repeat with ch in characters of restName
						set c to contents of ch
						if c is in "0123456789" then
							set digitsOnly to digitsOnly & c
						else
							exit repeat
						end if
					end repeat
					if digitsOnly is not "" then
						set v to digitsOnly as integer
						if v > maxSeq then set maxSeq to v
					end if
				end if
			end try
		end repeat
	end tell
	return maxSeq + 1
end nextSequenceInParent

-- 抓取随机封面并压缩到 ≤2MB
on fetchRandomCoverImage(coverPath)
	try
		set shellCmd to "/usr/bin/curl -Ls 'https://gallerix.asia/' | grep -Eo 'https?://[^\"'\\''']+\\.(jpg|jpeg|png|webp)' | grep '/p/' | sort -u | awk 'BEGIN{srand();} {a[NR]=$0} END{if(NR>0) print a[int(rand()*NR)+1]}'"
		set chosenURL to do shell script shellCmd
		
		if chosenURL is not "" then
			do shell script "/usr/bin/curl -Ls " & quoted form of chosenURL & " -o " & quoted form of coverPath
			-- 检查并压缩文件
			set fileSize to do shell script "stat -f%z " & quoted form of coverPath
			if (fileSize as integer) > 2000000 then
				do shell script "/usr/bin/sips --setProperty format jpeg --setProperty formatOptions 50 " & quoted form of coverPath & " >/dev/null 2>&1"
			end if
			return chosenURL
		else
			do shell script "/usr/bin/touch " & quoted form of coverPath
			return ""
		end if
	on error errMsg
		do shell script "/usr/bin/touch " & quoted form of coverPath
		return ""
	end try
end fetchRandomCoverImage

on buildFrontMatter(ymd, imgURL)
	set s to ""
	set s to s & "---" & linefeed
	set s to s & "title: \"这里是标题\"" & linefeed
	set s to s & "date: " & ymd & linefeed
	set s to s & "summary: \"这里是简介\"" & linefeed
	set s to s & "slug: " & linefeed
	set s to s & "draft: false" & linefeed
	set s to s & "image: \"cover.jpg\"" & linefeed
	set s to s & "thumbnail: \"cover.jpg\"" & linefeed
	if imgURL is not "" then set s to s & "image_url: \"" & imgURL & "\"" & linefeed
	set s to s & "categories: " & linefeed
	set s to s & "- 主题" & linefeed
	set s to s & "tags: " & linefeed
	set s to s & "- 标签" & linefeed
	set s to s & "---" & linefeed & linefeed
	set s to s & "在此输入正文…"
	return s
end buildFrontMatter

on run
	tell application id "DNtp"
		if not (exists think window 1) then activate
		set parentGroup to current group
		if parentGroup is missing value then set parentGroup to incoming group
		
		set dateStr to my todayStamp()
		set seq to my nextSequenceInParent(parentGroup, dateStr)
		set folderName to dateStr & "-" & (seq as string)
		
		set g to create record with {name:folderName, type:group} in parentGroup
		
		set tmpDir to do shell script "/usr/bin/mktemp -d"
		set indexPath to (tmpDir & "/" & indexFileName)
		set coverPath to (tmpDir & "/" & coverFileName)
		
		set ymd to my todayYMD()
		set imgURL to my fetchRandomCoverImage(coverPath)
		set fmText to my buildFrontMatter(ymd, imgURL)
		
		do shell script "/usr/bin/printf %s " & quoted form of fmText & " > " & quoted form of indexPath
		
		set idxRec to import path indexPath to g
		set covRec to import path coverPath to g
		
		try
			set current group of think window 1 to g
		end try
		activate
	end tell
end run
